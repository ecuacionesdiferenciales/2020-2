(x^{2}+y^{2})dy+2xydx=0
\end{equation}

\begin{equation}
\left(1+\left(\frac{y}{x}\right)^{2}\right)dy+2\left(\frac{y}{x}\right)dx=0
\end{equation} 
%Dividiendo entre $dx$
\begin{equation}
\left(1+\left(\frac{y}{x}\right)^{2}\right)\frac{dy}{dx}=-2\left(\frac{y}{x}\right)
\end{equation}
\begin{equation}
\frac{dy}{dx}=\frac{-2\left(\frac{y}{x}\right)}{1+\left(\frac{y}{x}\right)^{2}}
\end{equation}
